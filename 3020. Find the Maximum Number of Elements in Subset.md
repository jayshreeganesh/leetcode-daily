# LeetCode 3020: Find the Maximum Number of Elements in Subset

## Problem Description
Find the maximum number of elements in a subset that can be arranged in a symmetric "squaring chain" pattern: 
$$x, x^2, x^4, \dots, x^k, \dots, x^4, x^2, x$$

### Core Logic & Constraints
1. **Frequency Analysis**: Elements below the peak must appear $\ge 2$ times (left and right side of the symmetry). The peak element itself needs to appear $\ge 1$ time.
2. **Edge Case ($1$s)**: Since $1^2 = 1$, a sequence of $1$s can be arbitrary in length but must be an **odd length** to remain perfectly symmetric. For $c$ copies of $1$, the max length is $c$ (if odd) or $c-1$ (if even).
3. **Overflow Boundary**: The problem constraint caps values at $10^9$. If squaring an element exceeds $10^9$, that element can no longer have a follower and must serve as the peak element itself.

### 1. Go (Golang)
```go
package main

func maximumLength(nums []int) int {
    freq := make(map[int]int)
    for _, num := range nums {
        freq[num]++
    }

    maxLen := 0

    if countOne, exists := freq[1]; exists {
        if countOne%2 == 1 {
            maxLen = countOne
        } else {
            maxLen = countOne - 1
        }
    }

    for val := range freq {
        if val == 1 {
            continue
        }

        current := val
        currentLen := 0

        for freq[current] >= 2 {
            if current > 1e9/current {
                currentLen += 1
                current = -1
                break
            }
            currentLen += 2
            current = current * current
        }

        if current != -1 {
            if freq[current] >= 1 {
                currentLen += 1
            } else {
                currentLen -= 1
            }
        }

        if currentLen > maxLen {
            maxLen = currentLen
        }
    }

    return maxLen
}
```

### 2. Kotlin

```kotlin
import kotlin.math.max

class Solution {
    fun maximumLength(nums: IntArray): Int {
        val freq = HashMap<Long, Int>()
        for (num in nums) {
            val longNum = num.toLong()
            freq[longNum] = freq.getOrDefault(longNum, 0) + 1
        }

        var maxLen = 0

        if (freq.containsKey(1L)) {
            val countOne = freq[1L]!!
            maxLen = if (countOne % 2 == 1) countOne else countOne - 1
        }

        for (valKey in freq.keys) {
            if (valKey == 1L) continue

            var current = valKey
            var currentLen = 0
            var isPeakHandled = false

            while (freq.containsKey(current) && freq[current]!! >= 2) {
                if (current > 1_000_000_000L / current) {
                    currentLen += 1
                    isPeakHandled = true
                    break
                } else {
                    currentLen += 2
                    current = current * current
                }
            }

            if (!isPeakHandled) {
                if (freq.containsKey(current) && freq[current]!! >= 1) {
                    currentLen += 1
                } else {
                    currentLen -= 1
                }
            }

            maxLen = max(maxLen, currentLen)
        }

        return maxLen
    }
}
```

### 4. Rust

```rust
use std::collections::HashMap;
use std::cmp::max;

impl Solution {
    pub fn maximum_length(nums: Vec<i32>) -> i32 {
        let mut freq: HashMap<i64, i32> = HashMap::new();
        for num in nums {
            *freq.entry(num as i64).or_insert(0) += 1;
        }

        let mut max_len = 0;

        if let Some(&count_one) = freq.get(&1) {
            max_len = if count_one % 2 == 1 { count_one } else { count_one - 1 };
        }

        for (&val, _) in freq.iter() {
            if val == 1 { continue; }

            let mut current = val;
            let mut current_len = 0;
            let mut is_peak_handled = false;

            while let Some(&cnt) = freq.get(&current) {
                if cnt >= 2 {
                    if current > 1_000_000_000 / current {
                        current_len += 1;
                        is_peak_handled = true;
                        break;
                    }
                    current_len += 2;
                    current = current * current;
                } else {
                    break;
                }
            }

            if !is_peak_handled {
                if let Some(&cnt) = freq.get(&current) {
                    if cnt >= 1 { current_len += 1; } else { current_len -= 1; }
                } else {
                    current_len -= 1;
                }
            }

            max_len = max(max_len, current_len);
        }

        max_len
    }
}
```

### 9. Elixir

```elixir
defmodule Solution do
  @spec maximum_length(nums :: [integer]) :: integer
  def maximum_length(nums) do
    freq = Enum.frequencies(nums)

    max_len_ones = 
      case Map.get(freq, 1) do
        nil -> 0
        count -> if rem(count, 2) == 1, do: count, else: count - 1
      end

    Enum.reduce(freq, max_len_ones, fn {val, _count}, max_len ->
      if val == 1 do
        max_len
      else
        current_len = build_chain(val, freq, 0)
        max(max_len, current_len)
      end
    end)
  end

  defp build_chain(current, freq, len) do
    case Map.get(freq, current, 0) do
      count when count >= 2 ->
        if current > div(1_000_000_000, current) do
          len + 1
        else
          build_chain(current * current, freq, len + 2)
        end

      count when count == 1 ->
        len + 1

      _ ->
        len - 1
    end
  end
end
```

### 11. Racket

```racket
(define/contract (maximum-length nums)
  (-> (listof exact-integer?) exact-integer?)
  
  (define freq
    (for/fold ([acc (hash)])
              ([num (in-list nums)])
      (hash-update acc num add1 0)))

  (define max-len-ones
    (let ([count-one (hash-ref freq 1 0)])
      (if (zero? count-one)
          0
          (if (odd? count-one) count-one (- count-one 1)))))

  (define (build-chain current len)
    (let ([count (hash-ref freq current 0)])
      (cond
        [(>= count 2)
         (if (> current (quotient 1000000000 current))
             (+ len 1)
             (build-chain (* current current) (+ len 2)))]
        [(= count 1) (+ len 1)]
        [else (- len 1)])))

  (for/fold ([max-len max-len-ones])
            ([val (in-hash-keys freq)])
    (if (= val 1)
        max-len
        (max max-len (build-chain val 0)))))
```

## Complexity

* **Time Complexity**: $\mathcal{O}(N)$ where $N$ is the number of elements in `nums`. Populating the frequency maps takes $\mathcal{O}(N)$ linear time. The chains grow exponentially fast ($x \to x^2 \to x^4$), running at most $4$ to $5$ steps per sequence before crossing $10^9$.
* **Space Complexity**: $\mathcal{O}(N)$ to store frequencies in each language's native key-value lookup structures.
