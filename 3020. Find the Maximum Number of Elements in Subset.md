# LeetCode 3020: Find the Maximum Number of Elements in Subset (All 19 Languages)

## Problem Description
Find the maximum number of elements in a subset that can be arranged in a symmetric "squaring chain" pattern: 
$$x, x^2, x^4, \dots, x^k, \dots, x^4, x^2, x$$

---

### 1. C++
```cpp
#include <vector>
#include <unordered_map>
#include <algorithm>

using namespace std;

class Solution {
public:
    int maximumLength(vector<int>& nums) {
        unordered_map<long long, int> freq;
        for (int num : nums) {
            freq[num]++;
        }

        int max_len = 0;
        if (freq.count(1)) {
            int count_one = freq[1];
            max_len = (count_one % 2 == 1) ? count_one : count_one - 1;
        }

        for (auto& [val, count] : freq) {
            if (val == 1) continue;

            long long current = val;
            int current_len = 0;
            bool is_peak_handled = false;

            while (freq.count(current) && freq[current] >= 2) {
                if (current > 1000000000LL / current) {
                    current_len += 1;
                    is_peak_handled = true;
                    break;
                }
                current_len += 2;
                current = current * current;
            }

            if (!is_peak_handled) {
                if (freq.count(current) && freq[current] >= 1) {
                    current_len += 1;
                } else {
                    current_len -= 1;
                }
            }
            max_len = max(max_len, current_len);
        }
        return max_len;
    }
};
```

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// A basic structure for hash mapping in C
typedef struct {
    long long key;
    int count;
} HashEntry;

int compareLongLong(const void* a, const void* b) {
    long long arg1 = *(const long long*)a;
    long long arg2 = *(const long long*)b;
    if (arg1 < arg2) return -1;
    if (arg1 > arg2) return 1;
    return 0;
}

int maximumLength(int* nums, int numsSize) {
    long long* longNums = (long long*)malloc(numsSize * sizeof(long long));
    for (int i = 0; i < numsSize; i++) {
        longNums[i] = nums[i];
    }
    qsort(longNums, numsSize, sizeof(long long), compareLongLong);

    // Compress sorted array to extract unique counts
    HashEntry* freq = (HashEntry*)malloc(numsSize * sizeof(HashEntry));
    int uniqueSize = 0;
    
    for (int i = 0; i < numsSize; i++) {
        if (uniqueSize > 0 && freq[uniqueSize - 1].key == longNums[i]) {
            freq[uniqueSize - 1].count++;
        } else {
            freq[uniqueSize].key = longNums[i];
            freq[uniqueSize].count = 1;
            uniqueSize++;
        }
    }

    int maxLen = 0;
    int countOne = 0;
    for (int i = 0; i < uniqueSize; i++) {
        if (freq[i].key == 1) {
            countOne = freq[i].count;
            break;
        }
    }
    if (countOne > 0) {
        maxLen = (countOne % 2 == 1) ? countOne : countOne - 1;
    }

    // Lookup function simulating hash sets on compressed array
    int findCount(long long target, HashEntry* arr, int size) {
        int low = 0, high = size - 1;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (arr[mid].key == target) return arr[mid].count;
            if (arr[mid].key < target) low = mid + 1;
            else high = mid - 1;
        }
        return 0;
    }

    for (int i = 0; i < uniqueSize; i++) {
        if (freq[i].key == 1) continue;

        long long current = freq[i].key;
        int currentLen = 0;
        int isPeakHandled = 0;

        int currentCount = freq[i].count;
        while (currentCount >= 2) {
            if (current > 1000000000LL / current) {
                currentLen += 1;
                isPeakHandled = 1;
                break;
            }
            currentLen += 2;
            current = current * current;
            currentCount = findCount(current, freq, uniqueSize);
        }

        if (!isPeakHandled) {
            if (currentCount >= 1) {
                currentLen += 1;
            } else {
                currentLen -= 1;
            }
        }
        if (currentLen > maxLen) {
            maxLen = currentLen;
        }
    }

    free(longNums);
    free(freq);
    return maxLen;
}
```

```elixir
  def maximum_length(nums) do
    freq = Enum.frequencies(nums)

    max_len_ones = 
      case Map.get(freq, 1) do
        nil -> 0
        count -> if rem(count, 2) == 1, do: count, else: count - 1

      count when count >= 2 ->
        if current > div(1_000_000_000, current) do
          len + 1
        else
          build_chain(current * current, freq, len + 2)
        end

      count when count == 1 ->
        len + 1

      _ ->
        len - 1
    end
```

## Complexity

* **Time Complexity**: $\mathcal{O}(N)$ where $N$ is the length of `nums`. Generating the frequency maps runs in $\mathcal{O}(N)$ linear time. Due to the exponential nature of repeated squaring ($x \to x^2 \to x^4 \to x^8$), the loops process at most $4$ to $5$ nodes per distinct key before bounded limits are reached.
* **Space Complexity**: $\mathcal{O}(N)$ space to load map mappings across runtime memory bounds.
