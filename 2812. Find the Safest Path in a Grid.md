# Algorithmic Strategy Overview

To solve **Find the Safest Path in a Grid**, we must maximize the minimum Manhattan distance to any thief along our path from `(0, 0)` to `(n-1, n-1)`. This can be efficiently computed in two parts:

### Phase 1: Precomputing Thief Distances (Multi-Source BFS)

We initialize a distance matrix with `-1` and enqueue all thieves `(grid[r][c] == 1)` simultaneously into a queue with a starting distance of `0`. We run a standard Breadth-First Search (BFS). This fills our distance matrix such that every cell contains its exact Manhattan distance to the nearest thief in $O(N^2)$ time.

### Phase 2: Finding the Safest Path (Two Paradigms)

Depending on language paradigms and standard library architectures, we employ one of two optimal patterns for pathfinding:

1. **Max-Heap Dijkstra-Style (Greedy Selection):**
We use a Priority Queue (Max-Heap) initialized with the start cell. At each step, we pop the cell with the highest safeness factor seen so far. If it reaches the destination, that factor is guaranteed to be the maximum possible minimum safeness.
2. **Binary Search on the Answer (BS + BFS Validation):**
For languages where a Priority Queue/Heap is not built into the core standard library (such as JavaScript, Ruby, Elixir, Erlang, or Racket), we binary search over the possible safeness values $[0, 2N]$. For each middle value `mid`, we run a fast standard BFS to check if a valid path exists from start to end without stepping on cells with a distance less than `mid`.

# 1. C++23

```cpp
#include <vector>
#include <queue>
#include <algorithm>

class Solution {
public:
    int maximumSafenessFactor(std::vector<std::vector<int>>& grid) {
        int n = grid.size();
        
        // Quick exit if start or end is occupied by a thief
        if (grid[0][0] == 1 || grid[n - 1][n - 1] == 1) return 0;
        
        // Initialize distances matrix with -1
        std::vector<std::vector<int>> dist(n, std::vector<int>(n, -1));
        std::queue<std::pair<int, int>> q;
        
        // Push all thief positions into the multi-source queue
        for (int r = 0; r < n; ++r) {
            for (int c = 0; c < n; ++c) {
                if (grid[r][c] == 1) {
                    dist[r][c] = 0;
                    q.push({r, c});
                }
            }
        }
        
        // Direction vectors for moving up, down, left, right
        int dr[] = {-1, 1, 0, 0};
        int dc[] = {0, 0, -1, 1};
        
        // Calculate closest thief distance for every grid cell
        while (!q.empty()) {
            auto [r, c] = q.front();
            q.pop();
            
            for (int i = 0; i < 4; ++i) {
                int nr = r + dr[i];
                int nc = c + dc[i];
                
                if (nr >= 0 && nr < n && nc >= 0 && nc < n && dist[nr][nc] == -1) {
                    dist[nr][nc] = dist[r][c] + 1;
                    q.push({nr, nc});
                }
            }
        }
        
        // Dijkstra phase using a Max-Heap
        // Struct tracking: {safeness_factor, r, c}
        std::priority_queue<std::tuple<int, int, int>> pq;
        std::vector<std::vector<bool>> visited(n, std::vector<bool>(n, false));
        
        pq.push({dist[0][0], 0, 0});
        visited[0][0] = true;
        
        while (!pq.empty()) {
            auto [safeness, r, c] = pq.top();
            pq.pop();
            
            // Reached destination, return max min safeness found
            if (r == n - 1 && c == n - 1) return safeness;
            
            for (int i = 0; i < 4; ++i) {
                int nr = r + dr[i];
                int nc = c + dc[i];
                
                if (nr >= 0 && nr < n && nc >= 0 && nc < n && !visited[nr][nc]) {
                    visited[nr][nc] = true;
                    // Path safeness is governed by the bottleneck cell
                    pq.push({std::min(safeness, dist[nr][nc]), nr, nc});
                }
            }
        }
        
        return 0;
    }
};
```

```python
from collections import deque
import heapq
from typing import List

class Solution:
    def maximumSafenessFactor(self, grid: List[List[int]]) -> int:
        n = len(grid)
        
        if grid[0][0] == 1 or grid[n - 1][n - 1] == 1:
            return 0
            
        dist = [[-1] * n for _ in range(n)]
        q = deque()
        
        for r in range(n):
            for c in range(n):
                if grid[r][c] == 1:
                    dist[r][c] = 0
                    q.append((r, c))
                    
        dr = [-1, 1, 0, 0]
        dc = [0, 0, -1, 1]
        
        while q:
            r, c = q.popleft()
            for i in range(4):
                nr, nc = r + dr[i], c + dc[i]
                if 0 <= nr < n and 0 <= nc < n and dist[nr][nc] == -1:
                    dist[nr][nc] = dist[r][c] + 1
                    q.append((nr, nc))
                    
        # Max-Heap approach via value negation
        # Format: (-safeness, r, c)
        pq = [(-dist[0][0], 0, 0)]
        visited = [[False] * n for _ in range(n)]
        visited[0][0] = True
        
        while pq:
            neg_safeness, r, c = heapq.heappop(pq)
            safeness = -neg_safeness
            
            if r == n - 1 and c == n - 1:
                return safeness
                
            for i in range(4):
                nr, nc = r + dr[i], c + dc[i]
                if 0 <= nr < n and 0 <= nc < n and not visited[nr][nc]:
                    visited[nr][nc] = True
                    next_safeness = min(safeness, dist[nr][nc])
                    heapq.heappush(pq, (-next_safeness, nr, nc))
                    
        return 0
```

```go
import (

type Item struct {
	safeness, r, c int
}

type MaxHeap []*Item

func (h MaxHeap) Len() int           { return len(h) }
func (h MaxHeap) Less(i, j int) bool { return h[i].safeness > h[j].safeness } // Greater condition makes it a Max-Heap
func (h MaxHeap) Swap(i, j int)      { h[i], h[j] = h[j], h[i] }
func (h *MaxHeap) Push(x interface{}) { *h = append(*h, x.(*Item)) }
func (h *MaxHeap) Pop() interface{} {
	old := *h
	n := len(old)
	item := old[n-1]
	*h = old[0 : n-1]
	return item
}

	if grid[0][0] == 1 || grid[n-1][n-1] == 1 {
		return 0
	}

		dist[i] = make([]int, n)

		}

				dist[r][c] = 0

		}

		}
```

```dart
		}

class HeapNode {
  final int safeness;
  final int r;
  final int c;
  HeapNode(this.safeness, this.r, this.c);
}

class DartMaxHeap {
  final List<HeapNode> _heap = [];
  int get size => _heap.length;

  void push(HeapNode val) {
    _heap.add(val);
    _up(_heap.length - 1);
  }

  HeapNode? pop() {
    if (_heap.isEmpty) return null;
    HeapNode top = _heap[0];
    HeapNode bottom = _heap.removeLast();
    if (_heap.isNotEmpty) {
      _heap[0] = bottom;
      _down(0);
    }
    return top;
  }

  void _up(int i) {
    while (i > 0) {
      int p = (i - 1) ~/ 2;
      if (_heap[i].safeness <= _heap[p].safeness) break;
      HeapNode temp = _heap[i];
      _heap[i] = _heap[p];
      _heap[p] = temp;
      i = p;
    }
  }

  void _down(int i) {
    int len = _heap.length;
    while ((i * 2 + 1) < len) {
      int left = i * 2 + 1;
      int right = left + 1;
      int best = left;

      if (right < len && _heap[right].safeness > _heap[left].safeness) {
        best = right;
      }
      if (_heap[i].safeness >= _heap[best].safeness) break;
      HeapNode temp = _heap[i];
      _heap[i] = _heap[best];
      _heap[best] = temp;
      i = best;
    }
  }
}
```

```erlang
    {Q2, D2} = check_and_enqueue_ms(Q1, D1, R + 1, C, NextD, N),
    {Q3, D3} = check_and_enqueue_ms(Q2, D2, R, C - 1, NextD, N),
    check_and_enqueue_ms(Q3, D3, R, C + 1, NextD, N).

check_and_enqueue_ms(Q, Dist, NR, NC, NextD, N) ->
    if (NR >= 0) and (NR < N) and (NC >= 0) and (NC < N) ->
        case element(NC + 1, element(NR + 1, Dist)) == -1 of
            true -> {queue:in({NR, NC}, Q), put_2d(Dist, NR, NC, NextD)};
            false -> {Q, Dist}
        end;
    true -> {Q, Dist}
    end.

binary_search(Low, High, Dist, N) when Low =< High ->
    Mid = (Low + High) div 2,
    case is_valid_path(Mid, Dist, N) of
        true ->
            max(Mid, binary_search(Mid + 1, High, Dist, N));
        false ->
            binary_search(Low, Mid - 1, Dist, N)
    end;
binary_search(_Low, _High, _Dist, _N) -> -1.

is_valid_path(TargetSafeness, Dist, N) ->
    StartSafeness = element(1, element(1, Dist)),
    EndSafeness = element(N, element(N, Dist)),
    if (StartSafeness < TargetSafeness) orelse (EndSafeness < TargetSafeness) ->
        false;
    true ->
        Q = queue:in({0, 0}, queue:new()),
        Visited = create_2d_matrix(N, false),
        NewVisited = put_2d(Visited, 0, 0, true),
        check_path_bfs(Q, NewVisited, Dist, TargetSafeness, N)
    end.

check_path_bfs(Q, Visited, Dist, Target, N) ->
    case queue:out(Q) of
        {{value, {R, C}}, RestQ} ->
            if (R == N - 1) and (C == N - 1) -> true;
            true ->
                {NextQ, NextVisited} = process_dirs_path(RestQ, Visited, Dist, R, C, Target, N),
                check_path_bfs(NextQ, NextVisited, Dist, Target, N)
            end;
        {empty, _} ->
            false
    end.

process_dirs_path(Q, Visited, Dist, R, C, Target, N) ->
    {Q1, V1} = check_and_enqueue_path(Q, Visited, Dist, R - 1, C, Target, N),
    {Q2, V2} = check_and_enqueue_path(Q1, V1, Dist, R + 1, C, Target, N),
    {Q3, V3} = check_and_enqueue_path(Q2, V2, Dist, R, C - 1, Target, N),
    check_and_enqueue_path(Q3, V3, Dist, R, C + 1, Target, N).

check_and_enqueue_path(Q, Visited, Dist, NR, NC, Target, N) ->
    if (NR >= 0) and (NR < N) and (NC >= 0) and (NC < N) ->
        IsVisited = element(NC + 1, element(NR + 1, Visited)),
        CellSafeness = element(NC + 1, element(NR + 1, Dist)),
        if (not IsVisited) and (CellSafeness >= Target) ->
            {queue:in({NR, NC}, Q), put_2d(Visited, NR, NC, true)};
        true -> {Q, Visited}
        end;
    true -> {Q, Visited}
    end.

create_2d_matrix(N, Default) ->
    Row = list_to_tuple(lists:duplicate(N, Default)),
    list_to_tuple(lists:duplicate(N, Row)).

put_2d(Matrix, R, C, Val) ->
    Row = element(R + 1, Matrix),
    UpdatedRow = setelement(C + 1, Row, Val),
    setelement(R + 1, Matrix, UpdatedRow).
```

```racket
(define (make-data-queue) (mcons 'queue (mcons '() '())))

(define (enqueue! q val)
  (let* ([cell (mcons val '())]
         [ptr-pair (mcdr q)])
    (if (null? (mcar ptr-pair))
        (begin (set-mcar! ptr-pair cell) (set-mcdr! ptr-pair cell))
        (begin (set-mcdr! (mcdr ptr-pair) cell) (set-mcdr! ptr-pair cell)))))

(define (dequeue! q)
  (let* ([ptr-pair (mcdr q)]
         [head (mcar ptr-pair)])
    (if (null? head)
        (error "Empty Queue")
        (let ([val (mcar head)])
          (set-mcar! ptr-pair (mcdr head))
          (when (null? (mcar ptr-pair)) (set-mcdr! ptr-pair '()))
          val))))

(define (queue-empty? q) (null? (mcar (mcdr q))))
```
